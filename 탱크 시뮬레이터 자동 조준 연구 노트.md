# 변수 및 상수
적 위치 정보: dict(x: float, y: float, z: float)
내 위치 정보: dict(x: float, y: float, z: float)
적과의 거리: float
적 속도 정보: float
내 속도 정보: float
플레이 타임: float
오차 범위: float
포탑 최고 사거리: float = 127.78m
초기 탄속: float = 42.6.0m/s
적 이동 방향(헤딩값): float
내 이동 방향(헤딩값): float
입력 키값: dict("Q": float, "E": float, "R": float, "F": float)
적 바운딩 박스 중심 좌표: dict(x: float, y: float)

class 탄도학 {
    탄속: float = 42.6.0m/s
    중력 가속도: float = 9.81m/s^2
    거리 별 포신 각도: float = None

    function 거리 별 포신 각도 계산 함수(적과의 거리, 아군 y축, 적군 y축):
        if (적과의 거리 <= 포탑 최고 사거리) {
            높이 차 = 적군 y축 - 아군 y축
            시간 = 적과의 거리 / 탄속
            포신 각도 = arctan(높이 차 / 적과의 거리) + (0.5 * 중력 가속도 * 시간^2) / 탄속
            거리 별 포신 각도 = 포신 각도
            출력(거리 별 포신 각도)
        } else {
            출력("사거리 초과")
        }
}

class 조준 {
    계산된 포신 각도: float
    계산된 포탑 각도: float

    function 조준하기 행동_일반 {
        이미지 목표 = 적 바운딩 박스 중심 좌표
        시뮬레이션 목표 = 적 위치
        융합 목표 = (이미지 목표 + 시뮬레이션 목표) / 2
        계산된 포신 각도 = 일반제어.포신 각도를 오차 범위에 맞게 맞추기(융합 목표, 오차 범위)
        계산된 포탑 각도 = 일반제어.포탑 각도를 오차 범위에 맞게 맞추기(융합 목표, 오차 범위)
    }
    function 조준하기 행동_PID {
        PID 객체 불러오기
        계산된 포신 각도 = PID 제어.제어값 계산(오차 범위, 적 속도)
        계산된 포탑 각도 = PID 제어.제어값 계산(헤딩 오차)
    }
    function 조준하기 행동_MPC {
        MPC 객체 불러오기
        계산된 포신 각도 = MPC 제어.제어예측(적 위치, 플레이 타임)
        계산된 포탑 각도 = MPC 제어.제어예측(적 이동 방향)
    }
}

class 일반 제어 {
    function 포신 각도를 오차 범위에 맞게 맞추기(목표 각도, 오차 범위, 입력 키값) {
        현재 각도 = 포신 현재 각도
        오차 = 목표 각도 - 현재 각도
        if (abs(오차) <= 오차 범위) {
            포신 각도 = 목표 각도
        } else {
            if (입력 키값["Q"] > 0) 포신 각도 += 입력 키값["Q"]
            if (입력 키값["E"] > 0) 포신 각도 -= 입력 키값["E"]
        }
        return 포신 각도
    }
    function 포탑 각도를 오차 범위에 맞게 맞추기(목표 각도, 오차 범위, 입력 키값) {
        현재 각도 = 포탑 현재 각도
        오차 = 목표 각도 - 현재 각도
        if (abs(오차) <= 오차 범위) {
            포탑 각도 = 목표 각도
        } else {
            if (입력 키값["R"] > 0) 포탑 각도 += 입력 키값["R"]
            if (입력 키값["F"] > 0) 포탑 각도 -= 입력 키값["F"]
        }
        return 포탑 각도
    }
}

class PID 제어 {
    kp: float = 1.0
    ki: float = 0.1
    kd: float = 0.01
    이전 오차: float = 0
    적분 합: float = 0

    function 제어값 계산(오차, dt) {
        적분 합 += 오차 * dt
        미분 = (오차 - 이전 오차) / dt
        제어값 = kp * 오차 + ki * 적분 합 + kd * 미분
        이전 오차 = 오차
        return 제어값
    }
}

class MPC 제어 {
    포신 이동 속도: float = 0.5
    포탑 제어 속도: float = 0.3
    예상 제어 시간(delta_t): float = 0.1

    function 제어예측(목표 위치, 현재 시간) {
        예상 시간 동안 {
            포신 이동 = 포신 이동 속도 * delta_t
            포탑 이동 = 포탑 제어 속도 * delta_t
            if (abs(목표 위치 - 현재 위치) <= 오차 범위) {
                키보드 값 입력 = 0
            } else {
                키보드 값 입력 = 포신 이동 속도와 포탑 제어 속도를 고려한 값
            }
        }
        return 키보드 값 입력
    }
}
